void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    // Normalized pixel coordinates (from 0 to 1)
    vec2 uv = fragCoord/iResolution.xy - vec2(0.5, 0.5);
    float fAspectRatio = iResolution.x / iResolution.y;
    uv.x *= fAspectRatio;
    
    if (any(greaterThan(uv, vec2(0.5, 0.5))) || any(lessThan(uv, vec2(-0.5, -0.5))))
        fragColor = vec4(vec3(0.0), 1.0);
    else
    {
        float fLength = length(uv);
        float fRadius = 0.5;
        float fThicknessMin = 0.01;
        float fThickness = clamp(iMouse.x / iResolution.x, fThicknessMin, fRadius);
        float fFade = 0.001;
        
        float fOuterRadius = fRadius;
        float fInnerRadius = fRadius - fThickness;
        
        float firstStep = smoothstep(fInnerRadius, fInnerRadius + fFade, fLength);
        float secondStep = 1.0 - smoothstep(fOuterRadius, fOuterRadius + fFade, fLength);
        
        vec3 col = vec3(firstStep * secondStep);
        
        fragColor = vec4(col, 1.0);
    }
}
